#### 4. –ú–µ—Ç–æ–¥—ã –¥–ª—è –ø–æ–∏—Å–∫–∞ –≤—Å–µ—Ö –≤—Ö–æ–∂–¥–µ–Ω–∏–π
```python
# –ù–∞–π—Ç–∏ –≤—Å–µ –∏–Ω–¥–µ–∫—Å—ã —ç–ª–µ–º–µ–Ω—Ç–∞
def find_all(lst, value):
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö –∏–Ω–¥–µ–∫—Å–æ–≤, –≥–¥–µ –≤—Å—Ç—Ä–µ—á–∞–µ—Ç—Å—è value"""
    return [i for i, x in enumerate(lst) if x == value]

print(indices)  # [1, 3, 5]

# –ü–æ–¥—Å—á–µ—Ç –≤—Ö–æ–∂–¥–µ–Ω–∏–π —Å —É—Å–ª–æ–≤–∏–µ–º
def count_if(lst, condition):
    """–ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç —ç–ª–µ–º–µ–Ω—Ç—ã, —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è—é—â–∏–µ —É—Å–ª–æ–≤–∏—é"""
    return sum(1 for x in lst if condition(x))

numbers = [1, 5, 10, 15, 20, 25]
count = count_if(numbers, lambda x: x > 10)
print(count)  # 3

# –ù–∞–π—Ç–∏ –ø–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –ø–æ —É—Å–ª–æ–≤–∏—é
def find_first(lst, condition):
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç, —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è—é—â–∏–π —É—Å–ª–æ–≤–∏—é"""
    for item in lst:
        if condition(item):
            return item
    return None

```

```python
def remove_duplicates(lst):
    """–£–¥–∞–ª—è–µ—Ç –¥—É–±–ª–∏–∫–∞—Ç—ã, —Å–æ—Ö—Ä–∞–Ω—è—è –ø–æ—Ä—è–¥–æ–∫ –ø–µ—Ä–≤—ã—Ö –≤—Ö–æ–∂–¥–µ–Ω–∏–π"""
```

#### –ü—Ä–∏–º–µ—Ä 2: –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤

```python
def group_by(lst, key_func):
    """–ì—Ä—É–ø–ø–∏—Ä—É–µ—Ç —ç–ª–µ–º–µ–Ω—Ç—ã —Å–ø–∏—Å–∫–∞ –ø–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—É key_func"""
    groups = {}
    for item in lst:
        key = key_func(item)
        if key not in groups:
            groups[key] = []
        groups[key].append(item)
    return groups

# –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ —á–∏—Å–µ–ª –ø–æ —á–µ—Ç–Ω–æ—Å—Ç–∏

# {'odd': [1, 3, 5, 7], 'even': [2, 4, 6, 8]}

# –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ —Å–ª–æ–≤ –ø–æ –¥–ª–∏–Ω–µ
words = ['cat', 'dog', 'bird', 'elephant', 'ant']

# {3: ['cat', 'dog', 'ant'], 4: ['bird'], 8: ['elephant']}
```

#### –ü—Ä–∏–º–µ—Ä 3: –°–ª–∏—è–Ω–∏–µ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å–ø–∏—Å–∫–æ–≤
```python
def merge_sorted(list1, list2):
    """–û–±—ä–µ–¥–∏–Ω—è–µ—Ç –¥–≤–∞ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å–ø–∏—Å–∫–∞ –≤ –æ–¥–∏–Ω –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π"""
    result = []
    i, j = 0, 0
    
    while i < len(list1) and j < len(list2):
        if list1[i] <= list2[j]:
            result.append(list1[i])
            i += 1
        else:
            result.append(list2[j])
            j += 1
    
    # –î–æ–±–∞–≤–ª—è–µ–º –æ—Å—Ç–∞–≤—à–∏–µ—Å—è —ç–ª–µ–º–µ–Ω—Ç—ã
    result.extend(list1[i:])
    result.extend(list2[j:])
    
    return result

a = [1, 3, 5, 7]
b = [2, 4, 6, 8]
merged = merge_sorted(a, b)
print(merged)  # [1, 2, 3, 4, 5, 6, 7, 8]
```

#### –ü—Ä–∏–º–µ—Ä 4: –¶–∏–∫–ª–∏—á–µ—Å–∫–∏–π —Å–¥–≤–∏–≥ —Å–ø–∏—Å–∫–∞

```python
def rotate(lst, n):
    """–¶–∏–∫–ª–∏—á–µ—Å–∫–∏ —Å–¥–≤–∏–≥–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –Ω–∞ n –ø–æ–∑–∏—Ü–∏–π –≤–ø—Ä–∞–≤–æ"""
    if not lst:
        return lst
    n = n % len(lst)  # –û–±—Ä–∞–±–æ—Ç–∫–∞ –±–æ–ª—å—à–∏—Ö n
    return lst[-n:] + lst[:-n]

numbers = [1, 2, 3, 4, 5]
print(rotate(numbers, 2))   # [4, 5, 1, 2, 3]
print(rotate(numbers, -1))  # [2, 3, 4, 5, 1] (–≤–ª–µ–≤–æ)
print(rotate(numbers, 7))   # [4, 5, 1, 2, 3] (7 % 5 = 2)
```
