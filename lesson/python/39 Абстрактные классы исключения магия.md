- `__next__()` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ª–µ–¥—É—é—â–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –∏–ª–∏ –≤—ã–±—Ä–∞—Å—ã–≤–∞–µ—Ç `StopIteration`

–ü—Ä–∏–º–µ—Ä: –∏—Ç–µ—Ä–∞—Ç–æ—Ä —Å –Ω–∞—Ä–∞—Å—Ç–∞—é—â–µ–π —Å—É–º–º–æ–π
```py
class CumulativeSum:
    def __init__(self, numbers):
        self.numbers = numbers
        self.index = 0
        self.total = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.index >= len(self.numbers):
            raise StopIteration
        self.total += self.numbers[self.index]
        self.index += 1
        return self.total

data = [3, 5, 2, 4]
acc = CumulativeSum(data)
for value in acc:
    print(value)
```

    pass

        self.height = height

    def area(self):
